The data we are using has 6 target groups. Each target group can have instanes from 8 clusters.
For example, 'target_race' group looks like this:

"target_race":{
        "0":16,
        "1":13,
        "2":4,
        "3":13,
        "4":8,
        "5":4,
        "6":6,
        "7":4
    },

"0" -> "7" are the cluster labels and the value stored is the count of instances that belong to that cluster.

# we want to compute the percentage of each cluster in the target group
def get_percentages(dataset):

    # initialize an empty dictionary with keys that correspond to the target groups
    target_percentages = {target: None  for target in targets}

    # here we are iterating over the JSON file (called dataset)
    for target_group in dataset:

        # convert the current group to a list (the list has the count values)
        current_group = list(dataset[target_group])

        # find the sum of all the data points (this number tells us how many instances belong to that target group)
        sum_of_data_points = sum(current_group)

        if sum_of_data_points != 0:

            # find the percentage of each cluster in the current target group
            percentages = [(cluster / sum_of_data_points) * 100 for cluster in current_group]

            # store the percentages in the dictionary
            target_percentages[target_group] = percentages

    return target_percentages

# to create a chart in Altair we call Chart and pass the dataframe we just made
# to create a pie chart we call mark_arc()
alt.Chart(single).mark_arc().encode(

    # we pass the theta column and set stack to True
    theta=alt.Theta(field = 'theta', type = 'quantitative', stack = True),

    # we pass the color column so Altair knows based on what to color (in our case each cluster will have its own color)
    color=alt.Color(field = 'color', type = 'nominal', title='cluster'),

    # this will maintain the same order (according to theta)
    order=alt.Order(field="theta")
).properties(
        # add a title
        title = f'target_race',

        # set the width and height (change as appropriate)
        width = 150,
        height = 150
)

# mark text will allow us to display the percent values
text = base.mark_text(radius = 110).encode(

    # we just need 1 decimal placae
    text=alt.Text('theta:Q', format='.1f'),

    # show text when theta > 0
    color=alt.condition(
        alt.datum.theta > 0,
        alt.value("black"),
        alt.value(None)
    )
)

# we will store all the charts we make in a list
charts = []

# we will iterate over the target groups and do exactly what we did above but for each target group
for target in target_percentages:

    # we put these inside the loop because we will need a different color and theta for each chart
    single = target_percentages[target]
    theta = single
    color = num_clusters
    single = pd.DataFrame({"theta":theta, "color":color})

    # base chart we previously made
    base = alt.Chart(single).mark_arc().encode(
        theta=alt.Theta(field = 'theta', type = 'quantitative', stack = True),
        color=alt.Color(field = 'color', type = 'nominal', title='Cluster')
    )

    pie = base.mark_arc(outerRadius=100).encode(
        order = alt.Order(field="theta")
        ).properties(
        # we set the title to the current target
        title = f'{target}',
        width = 100,
        height = 100
    )

    # mark text will allow us to display the percent values
    text = base.mark_text(radius = 110).encode(

        # we just need 1 decimal placae
        text=alt.Text('theta:Q', format='.1f'),

        # show text when theta > 0
        color=alt.condition(
            alt.datum.theta > 0,
            alt.value("black"),
            alt.value(None)
        )
    )
    charts.append(pie + text)
